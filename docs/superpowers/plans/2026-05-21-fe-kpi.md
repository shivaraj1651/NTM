# fe-kpi: KPI Dashboard Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a standalone `/admin/kpi` page with two distinct sections — Business KRAs (tenant-level commitments) and Campaign KPIs (per-channel execution metrics) — plus a 30-day trend chart.

**Architecture:** New `KpiDashboardPage.tsx` reads from a `GET /api/v1/kpi/dashboard` endpoint (mocked via MSW). Section 1 renders KRA progress bars with RAG badges. Section 2 renders a recharts trend chart and a campaign breakdown table linking to existing `KpisPage`. `useKpiDashboard()` hook lives in a new `useKpi.ts` file.

**Tech Stack:** React 18, TypeScript, TanStack Query, recharts, MSW, Vitest, existing `DataTable` and Shadcn UI components

// frontend/src/hooks/useKpi.ts
import { useQuery } from '@tanstack/react-query'
import type { KpiDashboardData } from '@/types/admin'

async function getKpiDashboard(): Promise<KpiDashboardData> {
  const res = await fetch('/api/v1/kpi/dashboard')
  if (!res.ok) throw new Error('Failed to fetch KPI dashboard')
  return res.json()
}

export function useKpiDashboard() {
  return useQuery<KpiDashboardData>({
    queryKey: ['kpi-dashboard'],
    queryFn: getKpiDashboard,
  })
}
```

// frontend/src/pages/Admin/Kpi/KpiDashboardPage.tsx
import { useMemo } from 'react'
import { Link } from 'react-router-dom'
import { LineChart, Line, XAxis, YAxis, CartesianGrid, Tooltip, Legend, ResponsiveContainer } from 'recharts'
import type { ColumnDef } from '@tanstack/react-table'
import { DataTable } from '@/components/data-table'
import { Badge } from '@/components/ui/badge'
import { Card, CardContent, CardHeader, CardTitle } from '@/components/ui/card'
import { useKpiDashboard } from '@/hooks/useKpi'
import type { BusinessKra, CampaignKpiSummary } from '@/types/admin'
import { ArrowRight } from 'lucide-react'

// ─── RAG Alert Bar ───────────────────────────────────────────────────────────

function RagAlertBar({ red, amber, green }: { red: number; amber: number; green: number }) {
  return (
    <div className="flex gap-3 mb-6">
      <div className="flex items-center gap-2 rounded-md border border-red-200 bg-red-50 px-4 py-2">
        <span className="h-2 w-2 rounded-full bg-red-500" />
        <span className="text-sm font-medium text-red-700">Red: {red}</span>
      </div>
      <div className="flex items-center gap-2 rounded-md border border-amber-200 bg-amber-50 px-4 py-2">
        <span className="h-2 w-2 rounded-full bg-amber-500" />
        <span className="text-sm font-medium text-amber-700">Amber: {amber}</span>
      </div>
      <div className="flex items-center gap-2 rounded-md border border-green-200 bg-green-50 px-4 py-2">
        <span className="h-2 w-2 rounded-full bg-green-500" />
        <span className="text-sm font-medium text-green-700">Green: {green}</span>
      </div>
    </div>
  )
}

// ─── Section 1: Business KRAs ────────────────────────────────────────────────

function KraRow({ kra }: { kra: BusinessKra }) {
  const barColor =
    kra.status === 'green'
      ? 'bg-green-500'
      : kra.status === 'amber'
      ? 'bg-amber-500'
      : 'bg-red-500'

  const badge =
    kra.status === 'green' ? (
      <Badge className="bg-green-600 text-white text-xs">Green</Badge>
    ) : kra.status === 'amber' ? (
      <Badge variant="outline" className="border-amber-400 text-amber-600 text-xs">Amber</Badge>
    ) : (
      <Badge variant="destructive" className="text-xs">Red</Badge>
    )

  return (
    <div className="space-y-1">
      <div className="flex items-center justify-between">
        <span className="text-sm">{kra.description}</span>
        <div className="flex items-center gap-2">
          <span className="text-xs text-muted-foreground">
            {kra.actual} / {kra.target} {kra.unit}
          </span>
          {badge}
        </div>
      </div>
      <div className="h-2 rounded-full bg-muted overflow-hidden">
        <div
          className={`h-full rounded-full ${barColor} transition-all`}
          style={{ width: `${Math.min(kra.achievement_percent, 100)}%` }}
        />
      </div>
    </div>
  )
}

function KraSection({ kras }: { kras: BusinessKra[] }) {
  return (
    <Card className="mb-6">
      <CardHeader>
        <CardTitle className="text-base">Business KRAs</CardTitle>
      </CardHeader>
      <CardContent className="space-y-4">
        {kras.map((kra) => (
          <KraRow key={kra.kra_id} kra={kra} />
        ))}
      </CardContent>
    </Card>
  )
}

// ─── Section 2: Campaign KPIs ────────────────────────────────────────────────

function StatusBadge({ status }: { status: 'red' | 'amber' | 'green' }) {
  if (status === 'green') return <Badge className="bg-green-600 text-white text-xs">Green</Badge>
  if (status === 'amber') return <Badge variant="outline" className="border-amber-400 text-amber-600 text-xs">Amber</Badge>
  return <Badge variant="destructive" className="text-xs">Red</Badge>
}

function KpiSection({
  trends,
  campaigns,
}: {
  trends: { date: string; spend: number; impressions: number }[]
  campaigns: CampaignKpiSummary[]
}) {
  const columns = useMemo<ColumnDef<CampaignKpiSummary>[]>(
    () => [
      { accessorKey: 'campaign_id', header: 'Campaign' },
      {
        id: 'channels',
        header: 'Channels',
        cell: ({ row }) => (
          <div className="flex gap-1 flex-wrap">
            {row.original.channels.map((ch) => (
              <Badge key={ch} variant="secondary" className="text-xs">
                {ch.replace('_', ' ')}
              </Badge>
            ))}
          </div>
        ),
      },
      { accessorKey: 'top_kpi', header: 'Top KPI' },
      {
        id: 'spend',
        header: 'Spend',
        cell: ({ row }) => <span className="text-sm">${row.original.total_spend.toLocaleString()}</span>,
      },
      {
        id: 'status',
        header: 'Status',
        cell: ({ row }) => <StatusBadge status={row.original.overall_status} />,
      },
      {
        id: 'link',
        header: '',
        cell: ({ row }) => (
          <Link
            to={`/admin/campaigns/${row.original.campaign_id}/kpis`}
            className="flex items-center gap-1 text-xs text-muted-foreground hover:text-foreground"
          >
            Details <ArrowRight className="h-3 w-3" />
          </Link>
        ),
      },
    ],
    []
  )

  return (
    <Card>
      <CardHeader>
        <CardTitle className="text-base">Campaign KPIs — 30-Day Trends</CardTitle>
      </CardHeader>
      <CardContent className="space-y-6">
        <ResponsiveContainer width="100%" height={220}>
          <LineChart data={trends} margin={{ top: 4, right: 16, left: 0, bottom: 4 }}>
            <CartesianGrid strokeDasharray="3 3" className="stroke-muted" />
            <XAxis
              dataKey="date"
              tickFormatter={(d: string) => d.slice(5)}
              tick={{ fontSize: 11 }}
            />
            <YAxis yAxisId="left" tick={{ fontSize: 11 }} />
            <YAxis yAxisId="right" orientation="right" tick={{ fontSize: 11 }} />
            <Tooltip />
            <Legend />
            <Line yAxisId="left" type="monotone" dataKey="spend" stroke="#6366f1" dot={false} name="Spend ($)" />
            <Line yAxisId="right" type="monotone" dataKey="impressions" stroke="#22c55e" dot={false} name="Impressions" />
          </LineChart>
        </ResponsiveContainer>
        <DataTable columns={columns} data={campaigns} />
      </CardContent>
    </Card>
  )
}

// ─── Page Root ────────────────────────────────────────────────────────────────

export function KpiDashboardPage() {
  const { data, isLoading, isError } = useKpiDashboard()

  if (isLoading) return <p className="text-muted-foreground text-sm">Loading…</p>
  if (isError || !data) return <p className="text-destructive text-sm">Failed to load KPI dashboard.</p>

  return (
    <div className="space-y-6">
      <div>
        <h1 className="text-xl font-semibold">KPI Dashboard</h1>
        <p className="text-sm text-muted-foreground mt-1">
          Business commitments and campaign execution metrics across all campaigns.
        </p>
      </div>

      <RagAlertBar
        red={data.rag_summary.red}
        amber={data.rag_summary.amber}
        green={data.rag_summary.green}
      />

      <KraSection kras={data.kras} />

      <KpiSection trends={data.trends} campaigns={data.campaign_breakdown} />
    </div>
  )
}
```

// frontend/src/test/kpi-dashboard.test.tsx
import { render, screen } from '@testing-library/react'
import { MemoryRouter, Route, Routes } from 'react-router-dom'
import { QueryClient, QueryClientProvider } from '@tanstack/react-query'
import { describe, it, expect, beforeAll, afterEach, afterAll } from 'vitest'
import { setupServer } from 'msw/node'
import { kpiHandlers } from '@/mocks/handlers/kpi'
import { KpiDashboardPage } from '@/pages/Admin/Kpi/KpiDashboardPage'

const server = setupServer(...kpiHandlers)
beforeAll(() => server.listen())
afterEach(() => server.resetHandlers())
afterAll(() => server.close())

function renderPage() {
  const qc = new QueryClient({ defaultOptions: { queries: { retry: false } } })
  return render(
    <QueryClientProvider client={qc}>
      <MemoryRouter initialEntries={['/admin/kpi']}>
        <Routes>
          <Route path="/admin/kpi" element={<KpiDashboardPage />} />
        </Routes>
      </MemoryRouter>
    </QueryClientProvider>
  )
}

describe('KpiDashboardPage', () => {
  it('renders page title', async () => {
    renderPage()
    expect(await screen.findByText('KPI Dashboard')).toBeInTheDocument()
  })

  it('renders RAG summary bar with red, amber, green counts', async () => {
    renderPage()
    expect(await screen.findByText(/Red:/i)).toBeInTheDocument()
    expect(screen.getByText(/Amber:/i)).toBeInTheDocument()
    expect(screen.getByText(/Green:/i)).toBeInTheDocument()
  })

  it('renders Business KRAs section heading', async () => {
    renderPage()
    expect(await screen.findByText('Business KRAs')).toBeInTheDocument()
  })

  it('renders KRA descriptions', async () => {
    renderPage()
    expect(await screen.findByText(/Generate 500 leads in Hyderabad/i)).toBeInTheDocument()
    expect(screen.getByText(/Increase brand awareness by 20%/i)).toBeInTheDocument()
    expect(screen.getByText(/Reduce competitor share by 5 points/i)).toBeInTheDocument()
  })

  it('renders Campaign KPIs section with trend chart heading', async () => {
    renderPage()
    expect(await screen.findByText(/Campaign KPIs/i)).toBeInTheDocument()
  })

  it('renders campaign breakdown table rows', async () => {
    renderPage()
    expect(await screen.findByText('camp-001')).toBeInTheDocument()
    expect(screen.getByText('camp-002')).toBeInTheDocument()
    expect(screen.getByText('camp-003')).toBeInTheDocument()
  })

  it('campaign row links to campaign KPI page', async () => {
    renderPage()
    const links = await screen.findAllByRole('link', { name: /Details/i })
    expect(links[0]).toHaveAttribute('href', '/admin/campaigns/camp-001/kpis')
  })

  it('renders amber KRA badge for "Generate 500 leads"', async () => {
    renderPage()
    await screen.findByText(/Generate 500 leads/i)
    const amberBadges = screen.getAllByText('Amber')
    expect(amberBadges.length).toBeGreaterThan(0)
  })
})
