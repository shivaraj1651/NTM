# Audit Trail System Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a general-purpose `audit_trail` table + `AuditService` that captures all CRUD and agent events platform-wide, expose it via `/api/v1/admin/audit-trail`, and wire the existing `AuditLogPage.tsx` to the real API.

**Architecture:** Option B — `AuditMiddleware` (BaseHTTPMiddleware) extracts request context (actor_id, tenant_id, IP) into a `ContextVar`; `AuditService` injected via `Depends` reads that ContextVar and inserts rows; happy-path inserts share the caller's SQLAlchemy transaction, error-path inserts use a separate session so they commit even if the caller rolls back.

**Tech Stack:** Python 3.12, FastAPI, SQLAlchemy async (aiosqlite for tests), pytest-asyncio, React 18, TypeScript, TanStack Query, MSW 2.x

# backend/app/models/audit_trail.py
"""SQLAlchemy model for AuditTrail — insert-only general-purpose event log."""

from datetime import datetime, timezone
import uuid

from sqlalchemy import Column, DateTime, String, Text, Index, JSON
from sqlalchemy.orm import declarative_base

Base = declarative_base()


class AuditTrail(Base):
    __tablename__ = "audit_trail"

    id = Column(String, primary_key=True, default=lambda: str(uuid.uuid4()))
    tenant_id = Column(String, nullable=False)
    entity_type = Column(String, nullable=False)
    entity_id = Column(String, nullable=False)
    action = Column(String, nullable=False)
    # action values: create | read | update | delete | agent_run | tool_call
    actor_id = Column(String, nullable=False)
    actor_role = Column(String, nullable=True)
    ip_address = Column(String, nullable=True)
    user_agent = Column(String, nullable=True)
    request_path = Column(String, nullable=True)
    payload_before = Column(JSON, nullable=True)
    payload_after = Column(JSON, nullable=True)
    status = Column(String, nullable=False, default="success")
    # status values: success | error
    error_message = Column(Text, nullable=True)
    created_at = Column(
        DateTime(timezone=True),
        default=lambda: datetime.now(timezone.utc),
        nullable=False,
    )

    __table_args__ = (
        Index("ix_audit_trail_tenant", "tenant_id"),
        Index("ix_audit_trail_entity_type", "entity_type"),
        Index("ix_audit_trail_actor", "actor_id"),
        Index("ix_audit_trail_tenant_entity_type", "tenant_id", "entity_type"),
        Index("ix_audit_trail_tenant_created", "tenant_id", "created_at"),
    )

    def to_dict(self) -> dict:
        return {
            "id": self.id,
            "tenant_id": self.tenant_id,
            "entity_type": self.entity_type,
            "entity_id": self.entity_id,
            "action": self.action,
            "actor_id": self.actor_id,
            "actor_role": self.actor_role,
            "ip_address": self.ip_address,
            "user_agent": self.user_agent,
            "request_path": self.request_path,
            "payload_before": self.payload_before,
            "payload_after": self.payload_after,
            "status": self.status,
            "error_message": self.error_message,
            "created_at": self.created_at.isoformat() if self.created_at else None,
        }

# backend/app/core/audit_context.py
"""Per-request audit context stored in a ContextVar."""

from contextvars import ContextVar, Token

_audit_ctx: ContextVar[dict] = ContextVar("audit_ctx", default={})


def set_audit_context(data: dict) -> Token:
    """Set audit context data; returns token for later reset."""
    return _audit_ctx.set(data)


def get_audit_context() -> dict:
    """Get current audit context data."""
    return _audit_ctx.get()


def reset_audit_context(token: Token) -> None:
    """Restore context to state before set_audit_context was called."""
    _audit_ctx.reset(token)
```

# backend/app/core/audit_middleware.py
"""FastAPI middleware that sets per-request audit context from JWT + headers."""

import logging

from fastapi import Request
from jose import jwt, JWTError
from starlette.middleware.base import BaseHTTPMiddleware

from backend.app.core.audit_context import set_audit_context, reset_audit_context
from backend.app.core.config import settings

logger = logging.getLogger(__name__)


class AuditMiddleware(BaseHTTPMiddleware):
    """Extract actor/tenant/IP from every request and store in audit ContextVar."""

    async def dispatch(self, request: Request, call_next):
        actor_id = "anonymous"
        actor_role = None

        auth_header = request.headers.get("Authorization", "")
        if auth_header.startswith("Bearer "):
            raw_token = auth_header[len("Bearer "):]
            try:
                payload = jwt.decode(
                    raw_token,
                    settings.SECRET_KEY,
                    algorithms=[settings.ALGORITHM],
                )
                actor_id = payload.get("sub", "anonymous")
                actor_role = payload.get("role")
            except (JWTError, Exception):
                pass  # treat as anonymous

        forwarded_for = request.headers.get("X-Forwarded-For")
        ip_address = (
            forwarded_for.split(",")[0].strip()
            if forwarded_for
            else (request.client.host if request.client else "")
        )

        ctx_data = {
            "actor_id": actor_id,
            "actor_role": actor_role,
            "tenant_id": request.headers.get("X-Tenant-ID", ""),
            "ip_address": ip_address,
            "user_agent": request.headers.get("User-Agent", ""),
            "request_path": request.url.path,
        }
        token = set_audit_context(ctx_data)
        try:
            return await call_next(request)
        finally:
            reset_audit_context(token)
```

# backend/app/services/audit_service.py
"""AuditService — writes AuditTrail rows from router endpoints."""

import logging

from sqlalchemy.ext.asyncio import AsyncSession
from fastapi import Depends

from backend.app.db import get_db
from backend.app.models.audit_trail import AuditTrail
from backend.app.core.audit_context import get_audit_context

logger = logging.getLogger(__name__)


class AuditService:
    def __init__(self, db: AsyncSession) -> None:
        self._db = db

    async def emit(
        self,
        entity_type: str,
        entity_id: str,
        action: str,
        payload_before: dict | None = None,
        payload_after: dict | None = None,
        status: str = "success",
        error_message: str | None = None,
    ) -> None:
        """Insert one AuditTrail row. Never raises — logs errors internally."""
        try:
            ctx = get_audit_context()
            row = AuditTrail(
                tenant_id=ctx.get("tenant_id", ""),
                entity_type=entity_type,
                entity_id=str(entity_id),
                action=action,
                actor_id=ctx.get("actor_id", "anonymous"),
                actor_role=ctx.get("actor_role"),
                ip_address=ctx.get("ip_address"),
                user_agent=ctx.get("user_agent"),
                request_path=ctx.get("request_path"),
                payload_before=payload_before,
                payload_after=payload_after,
                status=status,
                error_message=error_message,
            )
            if status == "error":
                # Error-path: use a fresh session so the row commits
                # even if the caller's transaction rolled back.
                from backend.app.db import create_session_local
                async_session = create_session_local()
                async with async_session() as session:
                    session.add(row)
                    await session.commit()
            else:
                self._db.add(row)
                await self._db.flush()
        except Exception as exc:
            logger.warning("AuditService.emit failed silently: %s", exc)


async def get_audit_service(db: AsyncSession = Depends(get_db)) -> AuditService:
    """FastAPI dependency: provide AuditService bound to current DB session."""
    return AuditService(db)
```

"""Admin router — audit trail listing and detail (platform_admin only)."""

import logging
from datetime import datetime
from typing import Optional

from fastapi import APIRouter, Depends, HTTPException, Query
from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession

from backend.app.core.auth import current_user
from backend.app.core.models import User
from backend.app.db import get_db
from backend.app.models.audit_trail import AuditTrail
from backend.app.schemas.audit_trail import AuditTrailResponse

logger = logging.getLogger(__name__)

router = APIRouter(prefix="/api/v1/admin", tags=["audit"])


async def require_platform_admin(user: User = Depends(current_user)) -> User:
    if user.role.name != "platform_admin":
        raise HTTPException(status_code=403, detail="Platform admin access required")
    return user


@router.get("/audit-trail", response_model=list[AuditTrailResponse])
async def list_audit_trail(
    tenant_id: Optional[str] = Query(None),
    entity_type: Optional[str] = Query(None),
    actor_id: Optional[str] = Query(None),
    action: Optional[str] = Query(None),
    date_from: Optional[str] = Query(None),
    date_to: Optional[str] = Query(None),
    limit: int = Query(50, le=200),
    offset: int = Query(0),
    _: User = Depends(require_platform_admin),
    db: AsyncSession = Depends(get_db),
) -> list[AuditTrailResponse]:
    stmt = (
        select(AuditTrail)
        .order_by(AuditTrail.created_at.desc())
        .limit(limit)
        .offset(offset)
    )
    if tenant_id:
        stmt = stmt.where(AuditTrail.tenant_id == tenant_id)
    if entity_type:
        stmt = stmt.where(AuditTrail.entity_type == entity_type)
    if actor_id:
        stmt = stmt.where(AuditTrail.actor_id == actor_id)
    if action:
        stmt = stmt.where(AuditTrail.action == action)
    if date_from:
        stmt = stmt.where(AuditTrail.created_at >= datetime.fromisoformat(date_from))
    if date_to:
        stmt = stmt.where(AuditTrail.created_at <= datetime.fromisoformat(date_to))

    result = await db.execute(stmt)
    rows = result.scalars().all()
    return [AuditTrailResponse.model_validate(r) for r in rows]


@router.get("/audit-trail/{audit_id}", response_model=AuditTrailResponse)
async def get_audit_trail(
    audit_id: str,
    _: User = Depends(require_platform_admin),
    db: AsyncSession = Depends(get_db),
) -> AuditTrailResponse:
    result = await db.execute(
        select(AuditTrail).where(AuditTrail.id == audit_id)
    )
    row = result.scalar_one_or_none()
    if row is None:
        raise HTTPException(status_code=404, detail="Audit trail entry not found")
    return AuditTrailResponse.model_validate(row)
```
